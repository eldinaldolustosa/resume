% Macro for resume entries with PDF bookmark
\newcounter{cventry}
\newcommand{\cventry}[4]{%
  \refstepcounter{cventry}%
  \phantomsection%
  \pdfbookmark[2]{#1}{cventry-\thecventry}%
  \noindent
  \begin{tabularx}{\textwidth}{@{}>{\raggedright\arraybackslash}X >{\raggedleft\arraybackslash}X@{}}
    \textbf{#1} & #2 \\
    \textit{#3} & \textit{#4} \\
  \end{tabularx}
}
